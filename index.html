Iptables Rule for OpenVPN server cat <<'EOFipt' > /etc/openvpn/openvpn.bash #!/bin/bash PUBLIC_INET="$(ip -4 route ls | grep default | grep -Po '(?<=dev )(\S+)' | head -1)" IPCIDR='10.200.0.0/16' IPCIDR2='10.201.0.0/16' iptables -I FORWARD -s $IPCIDR -j ACCEPT iptables -I FORWARD -s $IPCIDR2 -j ACCEPT iptables -t nat -A POSTROUTING -o $PUBLIC_INET -j MASQUERADE iptables -t nat -A POSTROUTING -s $IPCIDR -o $PUBLIC_INET -j MASQUERADE iptables -t nat -A POSTROUTING -s $IPCIDR2 -o $PUBLIC_INET -j MASQUERADE EOFipt chmod +x /etc/openvpn/openvpn.bash bash /etc/openvpn/openvpn.bash #
